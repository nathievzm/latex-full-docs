\section*{Customizing bulleted lists}

\begin{verbatim}
    \newcounter{foo}
    \setcounter{foo}{5}    
\end{verbatim}

\newcounter{foo}
\setcounter{foo}{5}

\begin{itemize}
    \item \verb|\arabic{foo}| produces \arabic{foo}
    \item \verb|\roman{foo}| produces \roman{foo}
    \item \verb|\Roman{foo}| produces \Roman{foo}
    \item \verb|\Alph{foo}| produces \Alph{foo}
    \item \verb|\alph{foo}| produces \alph{foo}
\end{itemize}
